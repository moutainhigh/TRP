create or replace function cal_millisecond_time(mytime varchar2)
  return integer is
  result integer;
  resulttime integer;
begin
    select
(to_date(mytime,'yyyy-mm-dd hh:mi:ss')-to_date('1970-1-1','yyyy-mm-dd'))
*24*60*60*1000 into resulttime
from dual;
    result :=resulttime;
    return result;
end cal_millisecond_time;
